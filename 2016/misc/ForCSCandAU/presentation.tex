val sort: l:list int -> 
          Tot (m:list int{sorted m 
                          /\ (forall i. mem i l = mem i m)})
          (decreases (length l))
let rec sort l = match l with
  | [] -> []
  | pivot::tl ->
    let hi, lo = partition (cmp pivot) tl in
    let l' = append (sort lo) (pivot::sort hi) in
    dedup l' 
\end{pyglist}
