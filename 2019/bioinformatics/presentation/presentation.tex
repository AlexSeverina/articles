\begin{frame}[fragile]
  \transwipe[direction=90]
  \frametitle{Grammar}
\begin{verbatim}
s1: stem<s0>
any_str : any_smb*[2..10]
s0: any_str | any_str stem<s0> s0
any_smb: A | T | C | G
stem1<s>:               \\ stem of height exactly 1
      A s T | G s C | T s A | C s G
stem2<s>:               \\ stem of height exactly 2
      stem1< stem1<s> >
stem<s>:                \\ stem of height 3 or more
      A stem<s> T
    | T stem<s> A
    | C stem<s> G
    | G stem<s> C
    | stem1< stem2<s> >
\end{verbatim}

\end{frame}
