\begin{frame}
	\transwipe[direction=90]
	\frametitle{Задача}
	Разработка генератора синтаксических анализаторов для \bfseries{\underline {среды .NET}} \normalfont со следующими свойствами:  
	\begin{itemize}
		\item Работа с произвольными контекстно-свободными грамматиками
		\item Поддержка расширенных контекстно-свободных  грамматик
		\item Поддержка s-атрибутных грамматик
	\end{itemize}
\end{frame}

\begin{frame}
	\transwipe[direction=90]
	\frametitle{Алгоритм}
	 GLR-анализатор предназначен для работы с произвольной (\bfseries{\underline {в том числе неоднозначной!}} \normalfont) КС  грамматикой
	\begin{itemize}
		\item Работает за линейное время 
    \item {$O(n^{3})$ в худшем случае}
	\end{itemize}
	Рассмотренные подходы:
	\begin{itemize}
			\item Алгоритм Эрли				
			\item Алгоритм Томиты				
			\item Рекурсивно-восходящий алгоритм				
	\end{itemize}					
\end{frame}

\begin{frame}
	\transwipe[direction=90]
	\frametitle{Расширенные контекстно-свободные грамматики}
	Конструкции расширенных регулярных выражений в правых частях правил.
		\begin{itemize}
			\item удобно
			\item иногда необходимо
			\begin{itemize}
				\item повторение с указанием диапазона: x[1..5]
			  \item \bfseries{\underline {это не эквивалентно x^+}} \normalfont  		
			\end{itemize}
		\end{itemize}
	\end{frame}

\begin{frame}
	\transwipe[direction=90]
	\frametitle{Расширенные контекстно-свободные грамматики}
	Пример грамматики: 
		\begin{itemize}
			\item $S \rightarrow A(+A)*$
			\item $A \rightarrow a$
		\end{itemize}
	Преобразованная грамматика:
  \begin{itemize}
		\item $S \rightarrow AB$
		\item $A \rightarrow a$
		\item $B \rightarrow +AB$		
		\item $B \rightarrow \epsilon$
	\end{itemize}
	Входная цепочка: a+a

\end{frame}

\begin{frame}
	\transwipe[direction=90]
	\frametitle{Результаты}
	Реализован прототип GLR-анализатора, обладающий следующими свойствами:
	\begin{itemize}
		\item по однозначной LR-грамматике строится анализатор с линейной сложностью;
    \item по неоднозначной грамматике строится анализатор, возвращающий все возможные деревья вывода для данной входной цепочки;
    \item показана возможность поддержки расширенных КС грамматик;
	\end{itemize}	
\end{frame}
