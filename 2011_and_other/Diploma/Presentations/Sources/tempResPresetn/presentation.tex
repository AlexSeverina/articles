\begin{frame}
	\transwipe[direction=90]
	\frametitle{Задача}
	Разработка GLR-транслятора для среды .NET  со следующими свойствами:  
	\begin{itemize}
		\item Работа с произвольными КС грамматиками
		\item Поддержка расширенных контекстно-свободных  грамматик
		\item Поддержка s-атрибутных грамматик
	\end{itemize}
\end{frame}

\begin{frame}
	\transwipe[direction=90]
	\frametitle{Что такое GLR}
	 GLR-анализатор предназначен для работы с произвольной (в том числе неоднозначной!) КС  грамматикой
	\begin{itemize}
		\item Работает за линейное время 
    \item {$O(n^{3})$ в худшем случае}
	\end{itemize}	
\end{frame}

\begin{frame}
	\transwipe[direction=90]
	\frametitle{Область применения}
	Реинжиниринг программного обеспечения :
	\begin{itemize}
		\item Упрощение создания и сопровождения грамматик
		\begin{itemize}
			\item Нет необходимости задавать однозначную КС
			\item Нет десятков конфликтов при одном изменении
		\end{itemize}	
		\item Работа с диалектами одного языка    
		\begin{itemize}
			\item Задание общей грамматики
			\item Автоматическое определение диалекта
		\end{itemize}
	\end{itemize}
\end{frame}

\begin{frame}
	\transwipe[direction=90]
	\frametitle{Алгоритм трансляции}
	Требования к алгоритму:
		\begin{itemize}
		
			\item Поддержка произвольных КС грамматик
			\begin{itemize}
				\item Side-эффекты
			\end{itemize}
			
			\item Поддержка EBNF грамматик без их преобразования
			\begin{itemize}
				\item Правая часть правила - регулярное выражение
				\item Узлы одного уровня в дереве вывода - строка,порождённая регулярным выражением
			\end{itemize}

		\end{itemize}
\end{frame}

\begin{frame}
	\transwipe[direction=90]
	\frametitle{Алгоритм трансляции}
	Рассмотренные подходы:
		\begin{itemize}
		
			\item Отложенные вычисления (Continuation Passing Style, CPS)
			\begin{itemize}
				\item Конструирование функции, которая вычисляется после завершения вывода 
			\end{itemize}
			
			\item Интерпретация леса вывода
			\begin{itemize}
				\item Построение леса вывода
				\item Вычисление атрибутов
			\end{itemize}
		\end{itemize}
\end{frame}

\begin{frame}
	\transwipe[direction=90]
	\frametitle{Алгоритм трансляции.}
	Интерпретация леса вывода. Подходы:
		\begin{itemize}			
			\item Специальное преобразование грамматики на основе атрибутов
			\begin{itemize}		
				\item Атрибут - часть алфавита
				\item В дерево вывода добавляются узлы соответствующие атрибутам
				%\item $e \rightarrow (a{action1} b{action2}) {action3}$
			\end{itemize}			
			
			\item Преобразование дерева вывода во время вычисления атрибутов
			\begin{itemize}
				\item Листья - строка, порождённая регулярным выражением
				\item Построить дерево вывода этой строки
				\item Проинтерпретировать полученное дерево
			\end{itemize}

		\end{itemize}
\end{frame}
